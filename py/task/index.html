Программа сближающихся случайных чисел:
Установка matplotlib в командном интерпретаторе cmd: pip install matplotlib 
from random import random, randrange
from matplotlib import pyplot as plt

a, b = 1000, 3000
j = 0
c, d = [], []
e, f = 1, 9000000
while abs(a - b) > 1:
    a = randrange(e, f)
    b = randrange(e, f)
    e = min(a, b)
    f = max(a, b)
    j += 1
    c.append(j)
    d.append(abs(a - b))
    print("j", j, abs(a - b))

plt.plot(d, c, 'o')
plt.grid()
plt.show()

MKAD = 108.9
KAD = 142.15
road = KAD
v = float(input("v, км/ч: "))

s = v * t
x = s // road
g = s - x * road
if g - round(g):
    g += 1
print("Номер километра:", int(g))
